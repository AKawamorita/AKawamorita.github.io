    Responsável pelo pipeline de processamento e engenharia de atributos (Health Index).

    Esta classe transforma dados brutos de sensores em indicadores de saúde do ativo,

    def transform(self, dataset_root , run_paths):
        """
        Carrega e concatena múltiplos arquivos de dados (runs) em um único DataFrame.

        Args:
            dataset_root (str): Caminho raiz para os arquivos de dados.
            run_paths (dict): Dicionário mapeando IDs de run para seus respectivos caminhos.

        Returns:
            pd.DataFrame: DataFrame consolidado contendo todos os sinais dos sensores.
        """
        df = pd.DataFrame

        all_runs = []
        for run_id, run_folder in run_paths.items():
            print ("rodando ..."+run_id)
            df_run = textLoader.create_dataframe(run_folder, run_id)
            all_runs.append(df_run)

        df = pd.concat(all_runs, ignore_index=True)        

        # limpeza, conversões, validações
        return df


    def add_complementar_data(self, df_all):
        """
        Executa o pipeline completo de métricas complementares após o treino do modelo.

        Aplica sequencialmente o Health Index básico, zonas de aviso, 
        Health Index avançado (suavizado) e a lógica final de alertas.

        Note:
            Deve ser executado após a definição de HI_FEATURES pelo modelo principal.

        Args:
            df_all (pd.DataFrame): DataFrame original com as features extraídas.

        Returns:
            pd.DataFrame: DataFrame enriquecido com índices de saúde e estados de alerta.
        """
        df_all = self.add_health_index(
            df=df_all,
            run_col="run_id",
            feature_cols=HI_FEATURES,
            weights=[0.4, 0.3, 0.3],   # exemplo
            invert=True
        )

        df_all = self.add_warning_zones(
            df=df_all,
            hi_col="Health_Index",
            healthy_threshold=HEALTHY_THRESHOLD,
            warning_threshold=WARNING_THRESHOLD
        )

        df_hi = self.add_health_index_advanced(
            df=df_all,
            run_col="run_id",
            feature_cols=HI_FEATURES,
            weights=[0.45, 0.35, 0.20],
            invert=True,
            smooth_window=15
        )

        df_hi = self.add_alert_logic(
            df_hi,
            run_col="run_id",
            hi_col="Health_Index_Smoothed",
            baseline_points=100,
            slope_window=20,
            healthy_threshold=0.70,
            warning_threshold=0.40,
            early_drop_pct=0.08,
            warning_persistence=10,
            critical_persistence=5
        )
        return df_hi



    def add_warning_zones(self,
        df,
        hi_col="Health_Index",
        healthy_threshold=0.7,
        warning_threshold=0.4
    ):
        """
        Classifica as amostras em zonas de saúde baseadas em limiares estáticos.

        Cria categorias 'healthy', 'warning' e 'critical', além de flags binárias.

        Args:
            df (pd.DataFrame): DataFrame contendo a coluna de Health Index.
            hi_col (str, opcional): Nome da coluna de HI. Padrão "Health_Index".
            healthy_threshold (float, opcional): Limiar para zona saudável. Padrão 0.7.
            warning_threshold (float, opcional): Limiar para zona crítica. Padrão 0.4.

        Returns:
            pd.DataFrame: DataFrame com as novas colunas 'health_zone', 'warning_flag' e 'critical_flag'.
        """
        df_out = df.copy()

        def classify_zone(hi):
            if hi >= healthy_threshold:
                return "healthy"
            elif hi >= warning_threshold:
                return "warning"
            else:
                return "critical"

        df_out["health_zone"] = df_out[hi_col].apply(classify_zone)
        df_out["warning_flag"] = (df_out["health_zone"] != "healthy").astype(int)
        df_out["critical_flag"] = (df_out["health_zone"] == "critical").astype(int)

        return df_out

    def add_health_index(self,
        df,
        run_col="run_id",
        feature_cols=None,
        weights=None,
        invert=True
    ):
        """
        Calcula um Health Index (HI) básico através da combinação ponderada de features.

        Normaliza as features selecionadas por run (min-max) e aplica os pesos definidos.

        Args:
            df (pd.DataFrame): Dados de entrada.
            run_col (str): Coluna identificadora do experimento.
            feature_cols (list[str], opcional): Lista de features para o cálculo.
            weights (list[float], opcional): Pesos para cada feature. 
            invert (bool, opcional): Se True, 1 indica saudável e 0 indica falha.

        Returns:
            pd.DataFrame: DataFrame com a coluna 'Health_Index'.
        """
        df_out = df.copy()

        if feature_cols is None:
            feature_cols = ["s1_rms", "s1_kurtosis", "s1_crest_factor"]

        # Verifica se as colunas existem
        missing = [c for c in feature_cols if c not in df_out.columns]
        if missing:
            raise ValueError(f"Colunas ausentes no DataFrame: {missing}")

        # Se não houver pesos, usa média simples
        if weights is None:
            weights = np.ones(len(feature_cols)) / len(feature_cols)
        else:
            weights = np.array(weights, dtype=float)
            weights = weights / weights.sum()

        norm_cols = []

        # Normalização por run
        for col in feature_cols:
            norm_col = f"{col}_norm"
            norm_cols.append(norm_col)

            df_out[norm_col] = df_out.groupby(run_col)[col].transform(
                lambda x: (x - x.min()) / (x.max() - x.min() + 1e-8)
            )

        # Combinação ponderada
        df_out["HI_raw"] = 0.0
        for norm_col, w in zip(norm_cols, weights):
            df_out["HI_raw"] += df_out[norm_col] * w

        # Inverte: 1 = saudável, 0 = degradado
        if invert:
            df_out["Health_Index"] = 1 - df_out["HI_raw"]
        else:
            df_out["Health_Index"] = df_out["HI_raw"]

        return df_out

    def add_health_index_advanced(self,
        df,
        run_col="run_id",
        feature_cols=None,
        weights=None,
        invert=True,
        smooth_window=15
    ):
        """
        Versão avançada do Health Index com aplicação de média móvel para redução de ruído.

        Args:
            df (pd.DataFrame): Dados de entrada.
            run_col (str): Coluna identificadora do experimento.
            feature_cols (list[str], opcional): Lista de features.
            weights (list[float], opcional): Pesos ponderados.
            invert (bool): Inversão de escala (1=Saudável).
            smooth_window (int): Janela para a média móvel (rolling mean).

        Returns:
            pd.DataFrame: DataFrame com 'Health_Index_Smoothed'.
        """
        df_out = df.copy()

        if feature_cols is None:
            feature_cols = ["s3_std", "s3_var", "s4_shape_factor"]

        missing = [c for c in feature_cols if c not in df_out.columns]
        if missing:
            raise ValueError(f"Colunas ausentes no DataFrame: {missing}")

        if weights is None:
            weights = np.ones(len(feature_cols)) / len(feature_cols)
        else:
            weights = np.array(weights, dtype=float)
            weights = weights / weights.sum()

        norm_cols = []

        for col in feature_cols:
            norm_col = f"{col}_norm"
            df_out[norm_col] = df_out.groupby(run_col)[col].transform(
                lambda x: (x - x.min()) / (x.max() - x.min() + 1e-8)
            )
            norm_cols.append(norm_col)

        df_out["HI_raw"] = 0.0
        for w, c in zip(weights, norm_cols):
            df_out["HI_raw"] += w * df_out[c]

        if invert:
            df_out["Health_Index"] = 1 - df_out["HI_raw"]
        else:
            df_out["Health_Index"] = df_out["HI_raw"]

        # suavização por run
        df_out["Health_Index_Smoothed"] = df_out.groupby(run_col)["Health_Index"].transform(
            lambda x: x.rolling(window=smooth_window, min_periods=1).mean()
        )

        return df_out


    def add_alert_logic(self,
            df,
            run_col="run_id",
            hi_col="Health_Index_Smoothed",
            baseline_points=100,
            slope_window=20,
            healthy_threshold=0.70,
            warning_threshold=0.40,
            early_drop_pct=0.08,
            warning_persistence=10,
            critical_persistence=5
        ):
            """
            Lógica complexa para detecção de alertas baseada em queda brusca, inclinação e persistência.

            Avalia a saúde do ativo comparando-o com seu próprio baseline inicial e analisando
            a taxa de degradação (slope) e o tempo de permanência abaixo dos limiares.

            Args:
                df (pd.DataFrame): DataFrame com HI suavizado.
                run_col (str): Coluna de run.
                hi_col (str): Coluna do índice de saúde suavizado.
                baseline_points (int): Qtd. de pontos iniciais para considerar o ativo "novo".
                slope_window (int): Janela para cálculo da tendência (Regressão Linear).
                early_drop_pct (float): % de queda em relação ao baseline para alerta precoce.
                warning_persistence (int): Mínimo de pontos consecutivos para confirmar aviso.
                critical_persistence (int): Mínimo de pontos consecutivos para confirmar crítico.

            Returns:
                pd.DataFrame: DataFrame com a coluna 'alert_state' (healthy, early_warning, warning, critical).
            """
            df_out = df.copy()

            all_parts = []

            for run_id, temp in df_out.groupby(run_col):
                temp = temp.sort_values("time_index").copy()

                # baseline saudável inicial
                baseline = temp[hi_col].iloc[:baseline_points].mean()

                temp["baseline_hi"] = baseline
                temp["drop_pct"] = (baseline - temp[hi_col]) / (baseline + 1e-8)

                # slope
                temp["hi_slope"] = self.rolling_slope(temp[hi_col], window=slope_window)

                # condições
                temp["below_healthy"] = temp[hi_col] < healthy_threshold
                temp["below_warning"] = temp[hi_col] < warning_threshold
                temp["negative_slope"] = temp["hi_slope"] < 0
                temp["strong_negative_slope"] = temp["hi_slope"] < -0.002

                # persistência
                temp["below_healthy_count"] = self.consecutive_count(temp["below_healthy"])
                temp["below_warning_count"] = self.consecutive_count(temp["below_warning"])
                temp["negative_slope_count"] = self.consecutive_count(temp["negative_slope"])

                # alertas
                temp["early_warning"] = (
                    (temp["drop_pct"] >= early_drop_pct) &
                    (temp["negative_slope_count"] >= 5)
                ).astype(int)

                temp["warning"] = (
                    (temp["below_healthy_count"] >= warning_persistence)
                ).astype(int)

                temp["critical"] = (
                    (temp["below_warning_count"] >= critical_persistence) |
                    (temp["strong_negative_slope"])
                ).astype(int)

                def classify_alert(row):
                    if row["critical"] == 1:
                        return "critical"
                    elif row["warning"] == 1:
                        return "warning"
                    elif row["early_warning"] == 1:
                        return "early_warning"
                    else:
                        return "healthy"

                temp["alert_state"] = temp.apply(classify_alert, axis=1)

                all_parts.append(temp)

            return pd.concat(all_parts, axis=0)

    def rolling_slope(self, series, window=20):
        """
        Calcula a inclinação (slope) de uma série temporal usando Regressão Linear móvel.

        Args:
            series (pd.Series): Valores do Health Index.
            window (int): Janela de observação para a reta de regressão.

        Returns:
            pd.Series: Série com os coeficientes da inclinação.
        """
        values = series.values
        slopes = np.full(len(values), np.nan)

        x = np.arange(window).reshape(-1, 1)

        for i in range(window - 1, len(values)):
            y = values[i - window + 1:i + 1]
            model = LinearRegression()
            model.fit(x, y)
            slopes[i] = model.coef_[0]

        return pd.Series(slopes, index=series.index)

    def consecutive_count(self, condition_series):
        """
        Conta quantas vezes uma condição booleana foi verdadeira consecutivamente.

        Útil para evitar alarmes falsos causados por ruídos momentâneos.

        Args:
            condition_series (pd.Series): Série booleana (True/False).

        Returns:
            pd.Series: Contagem incremental dos verdadeiros.
        """

        count = np.zeros(len(condition_series), dtype=int)
        for i in range(len(condition_series)):
            if condition_series.iloc[i]:
                count[i] = count[i-1] + 1 if i > 0 else 1
            else:
                count[i] = 0
        return pd.Series(count, index=condition_series.index)


    def add_hi_raw (self,df_all):
        """Calcula o Health Index (HI) bruto baseado em normalização por grupo.

        Esta função realiza a normalização Min-Max das features especificadas em 
        `HI_FEATURES` dentro de cada `run_id`, calcula a média dessas features 
        normalizadas e inverte o resultado para gerar um índice de saúde onde 
        1 representa o estado saudável e 0 o estado degradado.

        Args:
            df_all (pd.DataFrame): DataFrame original contendo as colunas de 
                telemetria e o identificador de experimento 'run_id'.

        Returns:
            pd.DataFrame: Uma cópia do DataFrame original acrescida das colunas:
                - {feature}_norm: Versões normalizadas das HI_FEATURES.
                - HI_raw: Média simples das features normalizadas.
                - Health_Index: Indicador final (1 - HI_raw).
        """
        df_hi = df_all.copy()

        for col in HI_FEATURES:

            df_hi[col + "_norm"] = df_hi.groupby("run_id")[col].transform(
                lambda x: (x - x.min()) / (x.max() - x.min() + 1e-8)
            )

        df_hi["HI_raw"] = df_hi[[c + "_norm" for c in HI_FEATURES]].mean(axis=1)

            # 1 = saudável, 0 = degradado
        df_hi["Health_Index"] = 1 - df_hi["HI_raw"]

        return df_hi

            run_col (str): Coluna de run.
            hi_col (str): Coluna do índice de saúde suavizado.
            baseline_points (int): Qtd. de pontos iniciais para considerar o ativo "novo".
            slope_window (int): Janela para cálculo da tendência (Regressão Linear).
            early_drop_pct (float): % de queda em relação ao baseline para alerta precoce.
            warning_persistence (int): Mínimo de pontos consecutivos para confirmar aviso.
            critical_persistence (int): Mínimo de pontos consecutivos para confirmar crítico.

        Returns:
            pd.DataFrame: DataFrame com a coluna 'alert_state' (healthy, early_warning, warning, critical).
        """
        df_out = df.copy()

        all_parts = []

        for run_id, temp in df_out.groupby(run_col):
            temp = temp.sort_values("time_index").copy()

            # baseline saudável inicial
            baseline = temp[hi_col].iloc[:baseline_points].mean()

            temp["baseline_hi"] = baseline
            temp["drop_pct"] = (baseline - temp[hi_col]) / (baseline + 1e-8)

            # slope
            temp["hi_slope"] = self.rolling_slope(temp[hi_col], window=slope_window)

            # condições
            temp["below_healthy"] = temp[hi_col] < healthy_threshold
            temp["below_warning"] = temp[hi_col] < warning_threshold
            temp["negative_slope"] = temp["hi_slope"] < 0
            temp["strong_negative_slope"] = temp["hi_slope"] < -0.002

            # persistência
            temp["below_healthy_count"] = self.consecutive_count(temp["below_healthy"])
            temp["below_warning_count"] = self.consecutive_count(temp["below_warning"])
            temp["negative_slope_count"] = self.consecutive_count(temp["negative_slope"])

            # alertas
            temp["early_warning"] = (
                (temp["drop_pct"] >= early_drop_pct) &
                (temp["negative_slope_count"] >= 5)
            ).astype(int)

            temp["warning"] = (
                (temp["below_healthy_count"] >= warning_persistence)
            ).astype(int)

            temp["critical"] = (
                (temp["below_warning_count"] >= critical_persistence) |
                (temp["strong_negative_slope"])
            ).astype(int)

            def classify_alert(row):
                if row["critical"] == 1:
                    return "critical"
                elif row["warning"] == 1:
                    return "warning"
                elif row["early_warning"] == 1:
                    return "early_warning"
                else:
                    return "healthy"

            temp["alert_state"] = temp.apply(classify_alert, axis=1)

            all_parts.append(temp)

        return pd.concat(all_parts, axis=0)

    def add_health_index(self,
    df,
    run_col="run_id",
    feature_cols=None,
    weights=None,
    invert=True
):
    """
    Calcula um Health Index (HI) básico através da combinação ponderada de features.

    Normaliza as features selecionadas por run (min-max) e aplica os pesos definidos.

    Args:
        df (pd.DataFrame): Dados de entrada.
        run_col (str): Coluna identificadora do experimento.
        feature_cols (list[str], opcional): Lista de features para o cálculo.
        weights (list[float], opcional): Pesos para cada feature. 
        invert (bool, opcional): Se True, 1 indica saudável e 0 indica falha.

    Returns:
        pd.DataFrame: DataFrame com a coluna 'Health_Index'.
    """
    df_out = df.copy()

    if feature_cols is None:
        feature_cols = ["s1_rms", "s1_kurtosis", "s1_crest_factor"]

    # Verifica se as colunas existem
    missing = [c for c in feature_cols if c not in df_out.columns]
    if missing:
        raise ValueError(f"Colunas ausentes no DataFrame: {missing}")

    # Se não houver pesos, usa média simples
    if weights is None:
        weights = np.ones(len(feature_cols)) / len(feature_cols)
    else:
        weights = np.array(weights, dtype=float)
        weights = weights / weights.sum()

    norm_cols = []

    # Normalização por run
    for col in feature_cols:
        norm_col = f"{col}_norm"
        norm_cols.append(norm_col)

        df_out[norm_col] = df_out.groupby(run_col)[col].transform(
            lambda x: (x - x.min()) / (x.max() - x.min() + 1e-8)
        )

    # Combinação ponderada
    df_out["HI_raw"] = 0.0
    for norm_col, w in zip(norm_cols, weights):
        df_out["HI_raw"] += df_out[norm_col] * w

    # Inverte: 1 = saudável, 0 = degradado
    if invert:
        df_out["Health_Index"] = 1 - df_out["HI_raw"]
    else:
        df_out["Health_Index"] = df_out["HI_raw"]

    return df_out

    values = series.values

    x = np.arange(window).reshape(-1, 1)

        y = values[i - window + 1:i + 1]

        model.fit(x, y)
        slopes[i] = model.coef_[0]

    return pd.Series(slopes, index=series.index)

    def calcular_threshold_baseline(
    df_run: pd.DataFrame,
    coluna_sensor: str,
    baseline_pct: float = 0.20
) -> tuple[float, float, float]:
    """
    Calcula o threshold de alerta de uma feature com base no trecho inicial
    considerado saudável (baseline).

    Esta função utiliza os primeiros registros do DataFrame, definidos pelo
    percentual `baseline_pct`, para estimar o comportamento normal do sensor.
    A partir desse trecho, calcula a média e o desvio padrão da coluna
    informada e define o threshold pela regra estatística:

        threshold = media_normal + 3 * desvio_padrao_normal

    Args:
        df_run (pd.DataFrame): DataFrame contendo os dados de um único
            experimento, ativo ou `run_id`, preferencialmente já ordenados
            temporalmente.

        coluna_sensor (str): Nome da coluna numérica que será analisada para
            cálculo do threshold, como por exemplo `s3_std`, `s3_var` ou
            `s4_shape_factor`.

        baseline_pct (float, opcional): Percentual inicial do DataFrame a ser
            usado como região saudável (baseline). O valor padrão é `0.20`,
            ou seja, os primeiros 20% da série.

    Returns:
        tuple[float, float, float]:
            - media_normal (float): Média dos valores da feature no trecho
              saudável.
            - desvio_padrao_normal (float): Desvio padrão dos valores da
              feature no trecho saudável.
            - threshold (float): Valor de corte calculado pela fórmula
              `media_normal + 3 * desvio_padrao_normal`.

    Raises:
        TypeError: Se `df_run` não for um DataFrame do pandas.
        ValueError: Se o DataFrame estiver vazio.
        ValueError: Se `coluna_sensor` não existir no DataFrame.
        ValueError: Se `baseline_pct` não estiver entre 0 e 1.
        ValueError: Se a coluna do sensor não contiver valores numéricos válidos.

    Notes:
        - A função garante no mínimo 3 registros para o cálculo do baseline,
          mesmo que `baseline_pct` resulte em um valor muito pequeno.
        - Caso o desvio padrão resulte em `NaN` (por exemplo, quando houver
          poucos dados ou valores constantes), ele é ajustado para `0.0`.
        - Essa função assume que valores mais altos da feature representam
          maior degradação ou maior risco de falha.
        - Antes do cálculo, valores nulos da coluna analisada são removidos. """
    if not isinstance(df_run, pd.DataFrame):
        raise TypeError("df_run deve ser um pandas.DataFrame.")

    if df_run.empty:
        raise ValueError("df_run está vazio.")

    if coluna_sensor not in df_run.columns:
        raise ValueError(f"A coluna '{coluna_sensor}' não existe no DataFrame.")

    if not (0 < baseline_pct <= 1):
        raise ValueError("baseline_pct deve estar no intervalo (0, 1].")

    df_valid = df_run[[coluna_sensor]].copy()
    df_valid = df_valid.dropna()

    if df_valid.empty:
        raise ValueError(
            f"A coluna '{coluna_sensor}' não possui valores válidos para cálculo."
        )

    n_baseline = max(3, int(len(df_valid) * baseline_pct))
    n_baseline = min(n_baseline, len(df_valid))

    baseline = df_valid.iloc[:n_baseline]

    media_normal = baseline[coluna_sensor].mean()
    desvio_padrao_normal = baseline[coluna_sensor].std()

    if pd.isna(media_normal):
        raise ValueError(
            f"Não foi possível calcular a média para a coluna '{coluna_sensor}'."
        )

    if pd.isna(desvio_padrao_normal):
        desvio_padrao_normal = 0.0

    threshold = media_normal + 3 * desvio_padrao_normal

    return float(media_normal), float(desvio_padrao_normal), float(threshold)

    def localizar_alerta(df, run_id, coluna_sensor, window_size=10, baseline_pct=0.20):
    """
    Localiza o primeiro alerta de um sensor com threshold calculado automaticamente
    usando a regra: média_normal + 3 * desvio_padrao_normal.

    Args:
        df (pd.DataFrame): DataFrame completo.
        run_id (str): Identificador do experimento/grupo.
        coluna_sensor (str): Nome da coluna do sensor/feature.
        window_size (int): Tamanho da janela da média móvel.
        baseline_pct (float): Percentual inicial usado como regime saudável.

    Returns:
        dict: Resultado em formato JSON-friendly.
    """

    # 1. Filtra apenas o run desejado
    df_run = df[df["run_id"] == run_id].copy()

    if df_run.empty:
        return {
            "run_id": run_id,
            "sensor": coluna_sensor,
            "alerta_encontrado": False,
            "motivo": "run_id não encontrado",
            "threshold": None,
            "posicao_indice": None,
            "data_hora": "",
            "valor_detectado": None
        }

    if coluna_sensor not in df_run.columns:
        return {
            "run_id": run_id,
            "sensor": coluna_sensor,
            "alerta_encontrado": False,
            "motivo": "coluna_sensor não encontrada",
            "threshold": None,
            "posicao_indice": None,
            "data_hora": "",
            "valor_detectado": None
        }

    # 2. Ordena pelo índice temporal, se existir
    if "time_index" in df_run.columns:
        df_run = df_run.sort_values("time_index").copy()

    # 3. Define trecho saudável (primeiros X% do run)
    n_baseline = max(3, int(len(df_run) * baseline_pct))
    df_baseline = df_run.iloc[:n_baseline].copy()

    # 4. Calcula threshold = média_normal + 3 * desvio_padrao_normal
    media_normal, desvio_padrao_normal, threshold = calcular_threshold_baseline(
    df_baseline,
    coluna_sensor="s3_std",
    baseline_pct=0.20)

    # 5. Calcula média móvel no run inteiro
    moving_avg = df_run[coluna_sensor].rolling(window=window_size, min_periods=window_size).mean()

    # 6. Localiza alertas
    alertas = moving_avg[moving_avg > threshold]

    if not alertas.empty:
        primeiro_momento = alertas.index[0]
        posicao_na_lista = df_run.index.get_loc(primeiro_momento)

        data_hora = ""
        if "file_name" in df_run.columns:
            try:
                dt = pd.to_datetime(df_run.loc[primeiro_momento, "file_name"], format="%Y.%m.%d.%H.%M.%S")
                data_hora = dt.strftime("%d/%m/%Y %H:%M:%S")
            except:
                data_hora = str(df_run.loc[primeiro_momento, "file_name"])

        return {
            "run_id": run_id,
            "sensor": coluna_sensor,
            "alerta_encontrado": True,
            "motivo": "",
            "media_normal": round(media_normal, 6),
            "desvio_padrao_normal": round(desvio_padrao_normal, 6),
            "threshold": round(threshold, 6),
            "posicao_indice": int(posicao_na_lista),
            "data_hora": data_hora,
            "valor_detectado": round(float(alertas.iloc[0]), 6)
        }

    return {
        "run_id": run_id,
        "sensor": coluna_sensor,
        "alerta_encontrado": False,
        "motivo": "nenhum valor acima do threshold",
        "media_normal": round(media_normal, 6),
        "desvio_padrao_normal": round(desvio_padrao_normal, 6),
        "threshold": round(threshold, 6),
        "posicao_indice": None,
        "data_hora": "",
        "valor_detectado": None
    }
